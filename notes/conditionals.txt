To have a circle move from left to right and right to left and color changes based on mouseY:



circle.x = circle.x + circle.speed;

if (circle.x > width) {
    circle.speed = -circle.speed;
}

if (circle.x < 0 ) {
    circle.speed = -circle.speed;
}

if (mouseY < height/2) {
    fill (255,0,0);
}

if (mouseY > height/2) {
    fill (0,0,255);
}

ellipse(circle.x,circle.y,circle.size);
}
